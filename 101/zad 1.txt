Примерен вход: {1, 7, 10, 15, 16, 13, 23, 30} => Изход: 43 = 7 + 13 + 23*/
#include <stdio.h>

int prime(int a) {
    if (a <= 1){
        return false;
    }
    for (int i=2; i<a; i++){
        return false;
    }
    else (int sum) {
        sum = sum + a;
    }
}

    int arr[7] = 1, 7, 10, 15, 16, 13, 23, 30;
